# 1. Stop When Negative
# Write a program that continuously asks the user to enter numbers. 
# If the user enters a negative number, stop the loop using break. 
# Print the sum of all positive numbers entered.
sum = 0
while True:
    num = float(input('Enter a number: '))
    if num < 0:
        break
    sum += num
print('Sum of positive numbers:', sum)


# 2. Skip Multiples of 3
# Write a program that prints numbers from 1 to 20. 
# If a number is divisible by 3, skip it using continue.
n = int(input("Enter a number: "))

for i in range(0, n):
    if i % 3 == 0:
        continue
    print(i, end=' ')
